CREATE TABLE "DigitalTwin".city_park (
	id numeric NOT NULL,
	sigun_gu varchar(300) NULL,
	lon double precision NULL,
	lat double precision NULL,
	geom public.geometry(point, 4326) NULL,
	facility_type numeric NULL,
	CONSTRAINT city_park_pk PRIMARY KEY (id)
);
COMMENT ON TABLE "DigitalTwin".city_park IS '도시공원 테이블';

-- Column comments

COMMENT ON COLUMN "DigitalTwin".city_park.id IS '도시공원 고유 ID';
COMMENT ON COLUMN "DigitalTwin".city_park.sigun_gu IS '시군구 정보';
COMMENT ON COLUMN "DigitalTwin".city_park.lon IS '경도';
COMMENT ON COLUMN "DigitalTwin".city_park.lat IS '위도';
COMMENT ON COLUMN "DigitalTwin".city_park.geom IS 'geom 데이터';
COMMENT ON COLUMN "DigitalTwin".city_park.facility_type IS '시설물 타입 넘버';

UPDATE "DigitalTwin".city_park
SET geom = ST_SetSRID(
                ST_Point(lon, lat), 
                4326
           );
